Dataset Overview

The QM9 dataset is a comprehensive quantum chemistry database containing computed geometric, energetic, electronic, and thermodynamic properties for 134,000 stable small organic molecules made up of CHONF atoms (Carbon, Hydrogen, Oxygen, Nitrogen, Fluorine).

Each molecule contains up to 9 heavy (non-hydrogen) atoms and includes 19 regression targets derived from expensive DFT calculations. This dataset has become a standard benchmark for molecular property prediction in quantum machine learning.

What's Included

  • 134,000 organic molecules with up to 9 heavy atoms (C, N, O, F)
  • 3D molecular geometries optimized at B3LYP/6-31G(2df,p) level
  • 19 molecular properties including HOMO-LUMO gap, atomization energy, heat capacity
  • Quantum-encoded representations ready for VQE and QAOA algorithms
  • Pre-computed molecular Hamiltonians for quantum simulation
  • Benchmark splits for training, validation, and testing

Load QM9 Dataset with Qiskit 
# Install QDatasets package
pip install qdatasets

# Import libraries
from qdatasets import load_dataset
from qiskit_nature.second_q.mappers import JordanWignerMapper
from qiskit_nature.second_q.algorithms import GroundStateEigensolver

# Load QM9 dataset
qm9 = load_dataset('qm9-molecular', framework='qiskit')

# Access molecular data
molecules = qm9.get_molecules(limit=100)
properties = qm9.get_properties(['homo', 'lumo', 'gap'])

# Get quantum Hamiltonian for first molecule
hamiltonian = qm9.get_hamiltonian(0)

# Map to qubits using Jordan-Wigner
mapper = JordanWignerMapper()
qubit_op = mapper.map(hamiltonian)

print(f"Number of qubits required: {qubit_op.num_qubits}")
print(f"HOMO-LUMO gap: {properties['gap'][0]:.4f} eV")
Load QM9 Dataset with PennyLane 
# Install QDatasets and PennyLane
pip install qdatasets pennylane pennylane-qchem

# Import libraries
from qdatasets import load_dataset
import pennylane as qml
from pennylane import qchem

# Load QM9 dataset
qm9 = load_dataset('qm9-molecular', framework='pennylane')

# Get molecular data
molecule = qm9.get_molecule(0)
symbols, coordinates = molecule['atoms'], molecule['coords']

# Generate molecular Hamiltonian
H, qubits = qchem.molecular_hamiltonian(symbols, coordinates)

# Create quantum device
dev = qml.device('default.qubit', wires=qubits)

# Define VQE circuit
@qml.qnode(dev)
def circuit(params):
    qml.BasisState(molecule['hf_state'], wires=range(qubits))
    return qml.expval(H)

print(f"Molecule requires {qubits} qubits")
print(f"Ground state energy estimate: {circuit(params):.6f} Ha")
Load QM9 Dataset with Pure Python 
# Install QDatasets
pip install qdatasets pandas numpy

# Import libraries
from qdatasets import load_dataset
import pandas as pd
import numpy as np

# Load QM9 dataset as DataFrame
qm9 = load_dataset('qm9-molecular', format='pandas')
df = qm9.to_dataframe()

# Display basic statistics
print(f"Total molecules: {len(df)}")
print(f"Properties available: {df.columns.tolist()}")

# Filter molecules by criteria
small_gap = df[df['gap'] < 5.0]  # HOMO-LUMO gap < 5 eV
print(f"Molecules with gap < 5 eV: {len(small_gap)}")

# Get molecular properties
properties = ['homo', 'lumo', 'gap', 'zpve', 'cv']
subset = df[properties].head(10)

# Save filtered data
subset.to_csv('qm9_filtered.csv', index=False)
print("Data exported successfully!")
